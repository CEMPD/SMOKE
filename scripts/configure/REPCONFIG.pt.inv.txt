# This file is the report configuration file for running Smkreport for
# points sources for inventory-only reporting.

# All reports 
# Output file logical names are:
#	REPORT1: (P.1)  No spec, no gridding, by state
#	REPORT2: (P.2)  No spec, no gridding, by county
#	REPORT3: (P.3)  No spec, no gridding, by SCC
#	REPORT4: (P.4)  No spec, no gridding, by SCC and state
# 	REPORT5: (PG.1) No spec, with gridding, by state
#	REPORT6: (PG.3) No spec, with gridding,  by SCC

SMK_SOURCE P

/NEWFILE/ REPORT1

/CREATE REPORT/
    BY STATE NAME
    NUMBER F10.2
    UNITS ALL tons/day
/END/

/NEWFILE/ REPORT2

/CREATE REPORT/
    BY STATE NAME
    BY COUNTY NAME
    NUMBER F11.4
    UNITS ALL tons/day
/END/

/NEWFILE/ REPORT3

/CREATE REPORT/
    BY SCC10
    NUMBER F10.2
    UNITS ALL tons/day
/END/

/NEWFILE/ REPORT4

/CREATE REPORT/
    BY STATE NAME
    BY SCC10
    NUMBER F10.4
    UNITS ALL tons/day
/END/

/NEWFILE/ REPORT5

/CREATE REPORT/
    GRIDDING
    BY STATE NAME
    NUMBER F10.2
    UNITS ALL tons/day
/END/

/NEWFILE/ REPORT6

/CREATE REPORT/
    GRIDDING
    BY SCC10
    NUMBER F10.2
    UNITS ALL tons/day
/END/
