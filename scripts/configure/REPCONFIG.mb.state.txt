# This file is the report configuration file for running Smkreport for
# area sources for running state reports at each stage of processing.

# All reports 
# Output file logical names are:
#	REPORT1:    (M.1)  No spec, no gridding, no hourly by state
#	REPORT2:  (MGT.1)  No spec, with gridding, with hourly by state
#       REPORT3:  (MST.1)  With spec, no gridding, with hourly by state
#       REPORT4:   (MT.1)  No spec, no gridding, with hourly by state

SMK_SOURCE M

/NEWFILE/ REPORT1

/CREATE REPORT/
    BY STATE NAME
    NUMBER F10.1
    SELECT DATA VMT
    UNITS ALL miles/day
/END/

/NEWFILE/ REPORT2

/CREATE REPORT/
    BY STATE NAME
    GRIDDING
    TEMPORAL
    NUMBER F10.4
    UNITS ALL tons/day
/END/

/NEWFILE/ REPORT3

/CREATE REPORT/
    BY STATE NAME
    SPECIATION MASS
    TEMPORAL
    NUMBER F10.4
    SELECT DATA S-CO NO NO2 S-VOC
    UNITS ALL tons/day
/END/

/NEWFILE/ REPORT4

/CREATE REPORT/
    BY STATE NAME
    TEMPORAL
    NUMBER F10.4
    UNITS ALL tons/day
/END/
