# This file is the report configuration file for running Smkreport for
# area sources for inventory-only reporting.

# All reports 
# Output file logical names are:
#	REPORT1: (M.1)  No spec, no gridding, by state
#	REPORT2: (M.2)  No spec, no gridding, by county
#	REPORT3: (M.3)  No spec, no gridding, by SCC
#	REPORT4: (M.4)  No spec, no gridding, by SCC and state
#	REPORT5: (M.6)  No spec, no gridding, by Road class and state
# 	REPORT6: (MG.1) No spec, with gridding, by state
#	REPORT7: (MG.3) No spec, with gridding, by SCC

SMK_SOURCE M

/NEWFILE/ REPORT1
 
/CREATE REPORT/
    BY STATE NAME
    NUMBER F10.2
    UNITS ALL tons/day
/END/

/NEWFILE/ REPORT2

/CREATE REPORT/
    BY STATE NAME
    BY COUNTY NAME
    NUMBER F11.4
    UNITS ALL tons/day
/END/

/NEWFILE/ REPORT3

/CREATE REPORT/
    BY SCC10
    NUMBER F10.2
    UNITS ALL tons/day
/END/

/NEWFILE/ REPORT4

/CREATE REPORT/
    BY STATE NAME
    BY SCC10
    NUMBER F10.4
    UNITS ALL tons/day
/END/

/NEWFILE/ REPORT5

/CREATE REPORT/
    BY STATE NAME
    BY ROADCLASS
    NUMBER F10.3
    UNITS ALL tons/day
/END/

/NEWFILE/ REPORT6

/CREATE REPORT/
    GRIDDING
    BY STATE NAME
    NUMBER F10.2
    UNITS ALL tons/day
/END/

/NEWFILE/ REPORT7

/CREATE REPORT/
    GRIDDING
    BY SCC10
    NUMBER F10.2
    UNITS ALL tons/day
/END/
