# This file is the report configuration file for running Smkreport for
# mobile sources for running county reports at each stage of processing.

# All reports 
# Output file logical names are:
#	REPORT1:    (M.1)  No spec, no gridding, no hourly by county
#	REPORT1:  (MGT.1)  No spec, gridding, with hourly by county
#       REPORT3:  (MST.1)  With spec, no gridding, with hourly by county
#       REPORT4:   (MT.1)  No spec, no gridding, with hourly by county
#       REPORT5: (MGTS.1)  With spec, gridding, with hourly by county

SMK_SOURCE M

/NEWFILE/ REPORT1

/CREATE REPORT/
    BY STATE NAME
    BY COUNTY NAME
    NUMBER F12.4
    SELECT DATA VMT
    UNITS ALL tons/day
/END/

/NEWFILE/ REPORT2

/CREATE REPORT/
    BY STATE NAME
    BY COUNTY NAME
    GRIDDING
    TEMPORAL
    NUMBER F10.4
    UNITS ALL tons/day
/END/

/NEWFILE/ REPORT3

/CREATE REPORT/
    BY STATE NAME
    BY COUNTY NAME
    SPECIATION MASS
    TEMPORAL
    NUMBER F10.4
    UNITS ALL tons/day
/END/

/NEWFILE/ REPORT4

/CREATE REPORT/
    BY STATE NAME
    BY COUNTY NAME
    TEMPORAL
    NUMBER F10.4
    UNITS ALL tons/day
/END/

/NEWFILE/ REPORT5

/CREATE REPORT/
    BY STATE NAME
    BY COUNTY NAME
    TEMPORAL
    GRIDDING
    SPECIATION MASS
    NUMBER F10.4
    UNITS ALL tons/day
/END/
