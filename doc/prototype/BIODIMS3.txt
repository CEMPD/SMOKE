      INTEGER   NBFIP           !  max. # of FIPS codes
      INTEGER   MXVEGS          !  max. # of veg. type per land use category
      INTEGER   MXID            !  max. # of veg. id in emission factor table
      INTEGER   NSEF            !  # of em class in Summer emission factor table
      INTEGER   LUSES           !  dim:  land use ( gras, fore, wetl, agri
      INTEGER   BTYPES          !  dim:  VOC-emitting types
      INTEGER   TREETY          !  dim:  tree types (pine, decid, conif)
      INTEGER   GRAS            !  luse subscript for grass
      INTEGER   FORE            !  luse subscript for forest
      INTEGER   WETL            !  luse subscript for wetlands
      INTEGER   AGRI            !  luse subscript for agriculture
      INTEGER   RMTREE          !  luse subscript for agriculture
      INTEGER   SPTREE          !  luse subscript for agriculture
      REAL      FACTEN          !  m^2/hectare
 
      PARAMETER ( NBFIP    = 150, 
     &            MXVEGS   =  50, 
     &            MXID     = 200, 
     &            NSEF     =   4,
     &            LUSES    =   4, 
     &            BTYPES   =   6, 
     &            TREETY   =   3,
     &            GRAS     =   1, FORE = 2, WETL = 3, AGRI = 4,
     &            RMTREE   =   9, 
     &            SPTREE   =   5,
     &            FACTEN   =   1.0E4 )

C.......   Chemical-species-related parameters:

      INTEGER   BSPCS           !  dim for biogenic emissions species
      INTEGER   MSPCS           !  dim for model emissions species
      INTEGER   ISOP            !  bioem  subscript for ISOP
      INTEGER   MONO            !  bioem  subscript for MONO
      INTEGER   OVOC            !  bioem  subscript for OVOC
      INTEGER   NO              !  bioem  subscript for NO
      INTEGER   NNO             !  output em subscript for NO
      INTEGER   NPAR            !  output em subscript for NO
      INTEGER   NOLE            !  output em subscript for NO
      INTEGER   NALD            !  output em subscript for NO
      INTEGER   NISO            !  output em subscript for NO
      REAL      DENISO		!  inv mol wt for Isoprene
      REAL      DENMON		!  inv mol wt for Monoterpene
      REAL      DENUNK		!  inv mol wt for other VOC's
      REAL      DENNO		!  inv mol wt for NO
      REAL      MON2PAR         !  conversion of MON to PAR
      REAL      MON2OLE         !  conversion of MON to OLE
      REAL      MON2ALD2        !  conversion of MON to ALD2
      REAL      OTH2PAR         !  conversion of OTH to PAR
      REAL      OTH2OLE         !  conversion of OTH to OLE

      PARAMETER ( BSPCS    = 4, MSPCS  = 5 ,
     &            ISOP     = 1, MONO = 2, OVOC = 3, NO   = 4, 
     &            NNO      = 1, NPAR = 2, NOLE = 3, NALD = 4, NISO = 5,
     &            DENMON   = 1.0 / 136.24,    
     &            DENISO   = 1.0 /  68.12,
     &            DENUNK   = 1.0 / 148.00,
     &            DENNO    = 1.0 /  30.0 ,
     &            MON2PAR  = 6.0 * DENMON ,
     &            MON2OLE  = 0.5 * DENMON ,
     &            MON2ALD2 = 1.5 * DENMON ,
     &            OTH2PAR  = 8.5 * DENUNK ,
     &            OTH2OLE  = 0.5 * DENUNK )

C.......   Names for beis species, emitted model species:

        CHARACTER*4    BIOLUSE( LUSES )
        DATA            BIOLUSE / 'GRAS', 'FORE', 'WETL', 'AGRI' /

        CHARACTER*4    BIOLTYPE( BTYPES )
        DATA            BIOLTYPE / 
     &            'PINE', 'DECD', 'CONF', 'AGRC', 'LEAF', 'OTHR' /

        CHARACTER*16    BIOSPC( BSPCS )
        DATA            BIOSPC / 'ISOP', 'MONO', 'OVOC', 'NO' /

        CHARACTER*16    EMSPC( MSPCS )
        DATA            EMSPC / 'NO', 'PAR', 'OLE', 'ALD2', 'ISOP' /

        CHARACTER*4  OTHERID( RMTREE )
        DATA OTHERID / 'Mixf', 'Conf', 'Harf', 'Borf', 'Spin',
     &                 'Ufor', 'Oak', 'Odcd', 'Cnif' /

        CHARACTER*4  SPFORID( SPTREE )
        DATA  SPFORID / 'Borf', 'Wcnf', 'Wetf', 'Wmxf', 'Wwdl' /
